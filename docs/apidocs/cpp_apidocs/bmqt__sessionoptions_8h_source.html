485inline SessionOptions&
486SessionOptions::setChannelHighWatermark(bsls::Types::Int64 value)
487{
488 // PRECONDITIONS
489 BSLS_ASSERT_OPT(8 * 1024 * 1024 < value);
490 // We reserve 4MB for control message (see
491 // 'bmqimp::BrokerSession::k_CONTROL_DATA_WATERMARK_EXTRA') so make
492 // sure the provided value is greater than it.
493
494 d_channelHighWatermark = value;
495 return *this;
496}

498inline SessionOptions&
499SessionOptions::setStatsDumpInterval(const bsls::TimeInterval& value)
500{
501 // PRECONDITIONS
502 BSLS_ASSERT_OPT(value.seconds() % 30 == 0 &&
503 "value must be a multiple of 30s");
504 BSLS_ASSERT_OPT(0 <= value.seconds() && "value must be nonnegative");
505 BSLS_ASSERT_OPT(value.seconds() <
506 bdlt::TimeUnitRatio::k_SECONDS_PER_HOUR &&
507 "value must be < 60 min");
508
509 d_statsDumpInterval = value;
510 return *this;
511}
